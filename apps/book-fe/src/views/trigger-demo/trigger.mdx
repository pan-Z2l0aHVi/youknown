# Trigger

## 代码演示

import { Button, Divider, Space, Trigger } from '@youknown/react-ui/src'
import { useState } from 'react'

---

### 触发方式

export function TriggerMethods() {
  const [open, setOpen] = useState(false)
  const createContent = text => (
    <div className="w-max p-8px b-solid b-1 b-divider bg-#fff shadow-[1px_4px_8px_rgb(0,0,0,.1)]">{text}</div>
  )
  return (
    <Space>
      <Trigger trigger="hover" popup={createContent('tooltip')}>
        <Button>Hover me</Button>
      </Trigger>
      <Trigger trigger="click" popup={createContent('tooltip')}>
        <Button>Click me</Button>
      </Trigger>
      <Trigger
        trigger="manual"
        popup={createContent('tooltip')}
        open={open}
        onOpenChange={setOpen}
        onClickOutside={() => {
          setOpen(false)
        }}
      >
        <Button onClick={() => setOpen(true)}>Manual</Button>
      </Trigger>
    </Space>
  )
}

<TriggerMethods />

## API

### Trigger Props

| 属性            | 说明             | 类型                                                | 默认值           |
| --------------- | ---------------- | --------------------------------------------------- | ---------------- |
| defaultOpen     | 默认是否打开     | `boolean`                                           | `false`          |
| open            | 是否打开         | `boolean`                                           | -                |
| popup           | 弹出内容         | `React.ReactNode`                                   | -                |
| trigger         | 触发方式         | `'hover' \| 'click' \| 'manual'`                    | `'hover'`        |
| placement       | 排布             | `Placement`                                         | `'bottom-start'` |
| mouseEnterDelay | 移入事件延迟     | `number`                                            | `100`            |
| mouseLeaveDelay | 移除事件延迟     | `number`                                            | `100`            |
| spacing         | 间距偏移         | `number`                                            | `8`              |
| crossOffset     | 交叉偏移         | `number`                                            | `0`              |
| disabled        | 是否禁用         | `boolean`                                           | `false`          |
| unmountOnExit   | 关闭时卸载组件   | `boolean`                                           | `false`          |
| motion          | 动画             | `'none' \| 'grow' \| 'stretch' \| 'fade' \| 'zoom'` | `'none'`         |
| appendTo        | 插入到节点       | `HTMLElement \| null`                               | `document.body`  |
| onClickOutside  | 点击外部事件     | `(event: globalThis.MouseEvent) => void`            | -                |
| onOpenChange    | 打开变化回调函数 | `(open: boolean) => void`                           | -                |
